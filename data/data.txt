CREATE TABLE Customers (
    customerID INT PRIMARY KEY AUTO_INCREMENT,
    fullName VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    phoneNumber VARCHAR(15) NOT NULL,
    address TEXT NOT NULL,
    password VARCHAR(255) NOT NULL,
    registrationDate TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Bảng Nhân Viên
CREATE TABLE Employees (
    employeeID INT PRIMARY KEY AUTO_INCREMENT,
    fullName VARCHAR(100) NOT NULL,
    position VARCHAR(50) NOT NULL,
    phoneNumber VARCHAR(15) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL
);

-- Bảng Nhà Cung Cấp
CREATE TABLE Suppliers (
    supplierID INT PRIMARY KEY AUTO_INCREMENT,
    supplierName VARCHAR(100) NOT NULL,
    address TEXT NOT NULL,
    phoneNumber VARCHAR(15) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL
);


-- Bảng Danh Mục Sản Phẩm
CREATE TABLE ProductCategories (
    categoryID INT PRIMARY KEY AUTO_INCREMENT,
    categoryName VARCHAR(100) ,
    brandName VARCHAR(100) NOT NULL,
    seriesName VARCHAR(100),
    description TEXT
);

-- Bảng Sản Phẩm
CREATE TABLE Products (
    productID INT PRIMARY KEY AUTO_INCREMENT,
    productName VARCHAR(255) NOT NULL,
    categoryID INT,
    supplierID INT,
    description TEXT,
    price DECIMAL(10,2) NOT NULL,
    stockQuantity INT NOT NULL,
    availability ENUM('In Stock', 'Out of Stock') DEFAULT 'In Stock' NOT NULL,
    image VARCHAR(255),
    createdAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (categoryID) REFERENCES ProductCategories(categoryID),
 	 FOREIGN KEY (supplierID) REFERENCES Suppliers(supplierID)
);

-- Bảng Thuộc Tính Sản Phẩm (Attribute-Value Model)
CREATE TABLE ProductAttributes (
    attributeID INT PRIMARY KEY AUTO_INCREMENT,
    categoryID INT, 
    attributeName VARCHAR(100) NOT NULL,
    FOREIGN KEY (categoryID) REFERENCES ProductCategories(categoryID) ON DELETE CASCADE
);

-- Bảng Voucher (Giảm giá)
CREATE TABLE Vouchers (
    voucherID INT PRIMARY KEY AUTO_INCREMENT,
    code VARCHAR(50) UNIQUE NOT NULL,
    discount DECIMAL(5,2),
    expirationDate DATE
);

-- Bảng Đơn Hàng
CREATE TABLE Orders (
    orderID INT PRIMARY KEY AUTO_INCREMENT,
    customerID INT,
    employeeID INT,
    voucherID INT NULL,
    orderDate TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    totalAmount DECIMAL(10,2) NOT NULL,
    status ENUM('Pending', 'Shipping', 'Completed', 'Canceled') DEFAULT 'Pending',
    FOREIGN KEY (customerID) REFERENCES Customers(customerID),
    FOREIGN KEY (employeeID) REFERENCES Employees(employeeID),
    FOREIGN KEY (voucherID) REFERENCES Vouchers(voucherID) 
);

-- Bảng Chi Tiết Đơn Hàng
CREATE TABLE OrderDetails (
    orderDetailID INT PRIMARY KEY AUTO_INCREMENT,
    orderID INT,
    productID INT,
    quantity INT NOT NULL,
    unitPrice DECIMAL(10,2) NOT NULL,
    subtotal DECIMAL(10,2) NOT NULL,
    FOREIGN KEY (orderID) REFERENCES Orders(orderID) ON DELETE CASCADE,
    FOREIGN KEY (productID) REFERENCES Products(productID)
);



-- Bảng Phương Thức Thanh Toán
CREATE TABLE PaymentMethods (
    paymentID INT PRIMARY KEY AUTO_INCREMENT,
    methodName VARCHAR(100) UNIQUE NOT NULL
);

-- Bảng Thanh Toán
CREATE TABLE Payments (
    paymentID INT PRIMARY KEY AUTO_INCREMENT,
    orderID INT,
    paymentMethodID INT,
    paymentStatus ENUM('Pending', 'Completed', 'Failed') DEFAULT 'Pending',
    paymentDate TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    amount DECIMAL(10,2) NOT NULL,
    FOREIGN KEY (orderID) REFERENCES Orders(orderID) ON DELETE CASCADE,
    FOREIGN KEY (paymentMethodID) REFERENCES PaymentMethods(paymentID)
);

-- Bảng Đánh Giá Sản Phẩm
CREATE TABLE Reviews (
    reviewID INT PRIMARY KEY AUTO_INCREMENT,
    productID INT,
    customerID INT,
    rating INT CHECK (rating BETWEEN 1 AND 5),
    comment TEXT,
    reviewDate TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (productID) REFERENCES Products(productID) ON DELETE CASCADE,
    FOREIGN KEY (customerID) REFERENCES Customers(customerID) ON DELETE CASCADE
);

-- đặt lịch hẹn 
CREATE TABLE appointments (
    appointmentID INT PRIMARY KEY AUTO_INCREMENT,
    customerID INT NULL,
    employeeID INT NULL,
    appointmentDateTime DATETIME NOT NULL,
    estimatedArrivalTime DATETIME NOT NULL,
    duration INT DEFAULT 60,
    status ENUM('pending', 'confirmed', 'in_progress', 'completed', 'cancelled') DEFAULT 'pending',
    
    serviceType ENUM(
        'repair', 'assembly', 'installation', 'purchase', 
        'consultation', 'maintenance', 'upgrade', 
        'data_recovery', 'warranty_service', 'software_installation', 'other'
    ) NOT NULL,
    
    serviceLocation ENUM('home', 'office', 'store') NOT NULL DEFAULT 'store',
    
    deviceCategory ENUM(
        'BanPhim', 'Mouse', 'Mousepad', 'GamingGear', 'Headphone',
        'Case', 'CPU', 'Main', 'PSU', 'HDD', 'RAM', 'VGA', 
        'screen', 'PC', 'Laptop', 'iPad', 'Phone', 'other'
    ) NOT NULL,
    
    isWarrantyService BOOLEAN DEFAULT FALSE,
    warrantyPeriod INT NULL,
    
    address TEXT NULL,
    notes TEXT,
    estimatedCost DECIMAL(10,2) NULL,
    diagnosisDetails TEXT NULL,
    technicianNotes TEXT NULL,
    createdAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    guestName VARCHAR(100) NULL,
    guestEmail VARCHAR(100) NULL,
    guestPhone VARCHAR(20) NULL,

    FOREIGN KEY (customerID) REFERENCES customers(customerID) ON DELETE SET NULL,
    FOREIGN KEY (employeeID) REFERENCES employees(employeeID) ON DELETE SET NULL
);

-- trigger 
DELIMITER $$

CREATE TRIGGER trg_appointments_check_insert
BEFORE INSERT ON appointments
FOR EACH ROW
BEGIN
    IF NEW.customerID IS NULL AND (NEW.guestName IS NULL OR NEW.guestPhone IS NULL) THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Phải có customerID hoặc thông tin khách vãng lai (guestName và guestPhone)';
    END IF;

    IF NEW.serviceLocation <> 'store' AND NEW.address IS NULL THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Phải có địa chỉ nếu serviceLocation không phải là "store"';
    END IF;
END$$

DELIMITER ;

-- Trigger cho
DELIMITER $$

CREATE TRIGGER trg_appointments_check_update
BEFORE UPDATE ON appointments
FOR EACH ROW
BEGIN
    IF NEW.customerID IS NULL AND (NEW.guestName IS NULL OR NEW.guestPhone IS NULL) THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Phải có customerID hoặc thông tin khách vãng lai (guestName và guestPhone)';
    END IF;

    IF NEW.serviceLocation <> 'store' AND NEW.address IS NULL THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Phải có địa chỉ nếu serviceLocation không phải là "store"';
    END IF;
END$$

DELIMITER ;

-- 'in_stock' (có sẵn trong kho): Sau khi kiểm tra, phát hiện linh kiện đã có sẵn trong kho của cửa hàng và đã được trích xuất để chuẩn bị sử dụng cho lịch hẹn.
-- 'ordered' (đã đặt mua): Linh kiện không có sẵn trong kho và đã được đặt mua từ nhà cung cấp. Trạng thái này cho biết linh kiện đang trong quá trình vận chuyển hoặc chờ giao hàng.
